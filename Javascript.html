function Quote() {
  var Character = document.getElementById('Character').value;
  if (Character == "Hermione") {
    document.getElementById('quote').innerHTML = "We could all have been killed – or worse, expelled!";
  }
  else if (Character == "Dumbledore") {
    document.getElementById('quote').innerHTML = "Happiness can be found even in the darkest of times, if one only remembers to turn on the light.";
  }
  else if (Character == "Hagrid") {
    document.getElementById('quote').innerHTML = "You're a wizard, Harry.";
  }
  else if (Character == "Snape") {
    document.getElementById('quote').innerHTML = "Always.";
  }
  else {
	document.getElementById('quote').innerHTML = "Give me your wisdom...";
  }
}
