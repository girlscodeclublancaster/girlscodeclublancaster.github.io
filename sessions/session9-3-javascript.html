function Score() {
  var YourScore = 0;
  var YourName = document.getElementById('Question1').value;
  var Question2 = document.getElementById('Question2').value;
  if (Question2 == "CorrectAnswerOptionHere") {
    YourScore = YourScore + 1;
  }
  if (document.getElementById('Question3').checked) {
    YourScore = YourScore + 1;
  }
  var Question4 = document.getElementById('Question4').value;
  if (Question4 == 'CorrectAnswerOptionHere') {
    YourScore = YourScore + 1;
  }
  
  var Percent = Math.round((YourScore / 3) * 100);
  
  var Grade = "";
  if (Percent == 100) {
    Grade = "You must be a GENIUS, " + YourName + ". A+!";
  }
  else if (Percent > 89) {
    Grade = "So close to the top! Great job, " + YourName + ".";
  }
  else if (Percent > 79) {
    Grade = "Getting better. Keep it up, " + YourName + "!";
  }
  else if (Percent > 69) {
    Grade = "Good start, " + YourName + ". Try again!";
  }
  else if (Percent > 59) {
    Grade = "Not too great, " + YourName + ". Study some more and come back!";
  }
  else if (Percent < 59) {
    Grade = "Uh oh! More studying for you, " + YourName + ".";
  }
  var Message = "You scored " + Percent + "% (" + YourScore + " out of 3). " + Grade;
  
  document.getElementById('results').innerHTML = Message;
}
